**src/core/events.ts:**

Import `GameOutcomeSchema` from `./types.js`. Define event variants as separate Zod schemas, then combine into discriminated union:

```typescript
import { z } from 'zod'
import { GameOutcomeSchema } from './types.js'

export const ActionEventSchema = z.object({
  type: z.literal('action-taken'),
  gameId: z.string(),
  roundId: z.number().int(),
  playerId: z.string(),
  action: z.unknown(),
  reasoning: z.string().optional(),
  timestamp: z.string().datetime(),
})

export const PhaseChangedEventSchema = z.object({
  type: z.literal('phase-changed'),
  gameId: z.string(),
  from: z.string(),
  to: z.string(),
  timestamp: z.string().datetime(),
})

export const GameEndedEventSchema = z.object({
  type: z.literal('game-ended'),
  gameId: z.string(),
  outcome: GameOutcomeSchema,
  timestamp: z.string().datetime(),
})

export const CustomEventSchema = z.object({
  type: z.literal('custom'),
  gameId: z.string(),
  subtype: z.string(),
  payload: z.unknown(),
  timestamp: z.string().datetime(),
})

export const GameEventSchema = z.discriminatedUnion('type', [
  ActionEventSchema,
  PhaseChangedEventSchema,
  GameEndedEventSchema,
  CustomEventSchema,
])

export type GameEvent = z.infer<typeof GameEventSchema>
export type ActionEvent = z.infer<typeof ActionEventSchema>
export type PhaseChangedEvent = z.infer<typeof PhaseChangedEventSchema>
export type GameEndedEvent = z.infer<typeof GameEndedEventSchema>
export type CustomEvent = z.infer<typeof CustomEventSchema>
```

Key decisions per CONTEXT.md:
- One event per atomic action (fine-grained)
- Single `reasoning` string field on action events
- Event schema is game-agnostic — no Avalon fields
- `custom` variant with `subtype` + `payload` for game-specific extensions (per RESEARCH.md open question 1)
- `action` field uses `z.unknown()` (NOT `z.any()`) per pitfall 3

**src/core/event-bus.ts:**

Implement typed pub/sub. Use `Map<string, Set<Handler>>`. The public API:
- `emit(event: GameEvent)`: dispatches to all handlers registered for `event.type`
- `on<T>(type, handler)`: registers handler with TypeScript narrowing via `Extract<GameEvent, { type: T }>`
- `off<T>(type, handler)`: removes a specific handler

```typescript
import type { GameEvent } from './events.js'

type Handler<T extends GameEvent = GameEvent> = (event: T) => void

export class EventBus {
  private handlers = new Map<string, Set<Handler>>()

  emit(event: GameEvent): void {
    const hs = this.handlers.get(event.type)
    if (hs) for (const h of hs) h(event)
  }

  on<T extends GameEvent['type']>(
    type: T,
    handler: Handler<Extract<GameEvent, { type: T }>>,
  ): void {
    if (!this.handlers.has(type)) this.handlers.set(type, new Set())
    this.handlers.get(type)!.add(handler as Handler)
  }

  off<T extends GameEvent['type']>(
    type: T,
    handler: Handler<Extract<GameEvent, { type: T }>>,
  ): void {
    this.handlers.get(type)?.delete(handler as Handler)
  }
}
```

No game-specific code. No `await` in handlers. Fire-and-forget dispatch.

**src/logging/recorder.ts:**

Subscribe to EventBus events and write JSONL via Pino child logger. Per RESEARCH.md:
- Create root Pino logger with `pino.destination(filePath)` for async buffered writes
- Create child logger with `{ gameId }` auto-stamped on every line
- Subscribe to all event types and call `childLogger.info(event)` for each
- Provide a `flush()` method that calls `pino.destination.flushSync()` for test teardown
- Do NOT use `fs.writeFileSync` (per pitfall 5)

```typescript
import pino from 'pino'
import type { EventBus } from '../core/event-bus.js'

export class Recorder {
  private logger: pino.Logger
  private destination: pino.DestinationStream

  constructor(eventBus: EventBus, gameId: string, filePath: string) {
    this.destination = pino.destination(filePath)
    const root = pino({ level: 'info' }, this.destination)
    this.logger = root.child({ gameId })

    eventBus.on('action-taken', (e) => this.logger.info(e))
    eventBus.on('phase-changed', (e) => this.logger.info(e))
    eventBus.on('game-ended', (e) => this.logger.info(e))
    eventBus.on('custom', (e) => this.logger.info(e))
  }

  flush(): void {
    if ('flushSync' in this.destination) {
      (this.destination as pino.DestinationStream & { flushSync: () => void }).flushSync()
    }
  }
}
```
